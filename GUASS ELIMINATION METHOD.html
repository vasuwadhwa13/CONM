#include <stdio.h>

#define SIZE 10 // Maximum size of the system of equations

// Function to perform Gauss Elimination
void gaussElimination(float a[SIZE][SIZE], int n) {
    int i, j, k;
    float ratio;

    // Forward elimination
    for (i = 0; i < n - 1; i++) {
        for (j = i + 1; j < n; j++) {
            ratio = a[j][i] / a[i][i];
            for (k = 0; k <= n; k++) {
                a[j][k] = a[j][k] - ratio * a[i][k];
            }
        }
    }

    // Back substitution
    float x[SIZE];
    x[n - 1] = a[n - 1][n] / a[n - 1][n - 1];
    for (i = n - 2; i >= 0; i--) {
        x[i] = a[i][n];
        for (j = i + 1; j < n; j++) {
            x[i] = x[i] - a[i][j] * x[j];
        }
        x[i] = x[i] / a[i][i];
    }

    // Display solution
    printf("\nSolution:\n");
    for (i = 0; i < n; i++) {
        printf("x[%d] = %.2f\n", i, x[i]);
    }
}

int main() {
    float a[SIZE][SIZE];
    int n, i, j;

    // Input number of unknowns
    printf("Enter the number of unknowns: ");
    scanf("%d", &n);

    // Input augmented matrix
    printf("Enter the elements of the augmented matrix row by row:\n");
    for (i = 0; i < n; i++) {
        for (j = 0; j <= n; j++) {
            scanf("%f", &a[i][j]);
        }
    }

    // Perform Gauss Elimination
    gaussElimination(a, n);

    return 0;
}
