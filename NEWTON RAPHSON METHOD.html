Newton Raphson

Code


#include<stdio.h>
#include<math.h>

// Function whose root we are trying to find
float func(float x) {
    return x*x*x - x*x + 2; // Example: x^3 - x^2 + 2
}

// Derivative of the above function
float derivativeFunc(float x) {
    return 3*x*x - 2*x; // Derivative of x^3 - x^2 + 2
}

// Newton-Raphson method
void newtonRaphson(float x) {
    float h = func(x) / derivativeFunc(x);
    int count = 1;
    // Loop until the value of h becomes small enough (convergence condition)
    while (fabs(h) >= 0.0001) {
        h = func(x) / derivativeFunc(x);
        
        // x(i+1) = x(i) - f(x) / f'(x)
        x = x - h;
        
        printf("Iteration %d : %f\n",count, x);
        count++;
    }
    
    printf("The root is: %f\n", x);
}

int main() {
    float x0; // Initial guess
    printf("Enter the initial guess: ");
    scanf("%f", &x0);

    newtonRaphson(x0);    
    return 0;
}
